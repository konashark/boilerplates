var APIS = APIS || {};

APIS.STORAGE = (function(undefined) {
    var NOT_SUPPORTED_MSG = "Local Storage not supported on this device.";

    
    var persistentWrite = function(name, value) {
        if(typeof(localStorage) == "object") {
            localStorage.setItem(name, JSON.stringify(value));
        } else {
            console.log(NOT_SUPPORTED_MSG);
        }
    };
    
    var persistentRead = function(name) {
        if(typeof(localStorage) == "object") {
            var value = localStorage.getItem(name);
            if (value !== "undefined") {
                value = JSON.parse(value);
            }
            return value;
        } else {
            console.log(NOT_SUPPORTED_MSG);
        }
    };

    var persistentRemove = function(name) {
        console.log("LOCAL STORAGE: Removing " + name);
        if(typeof(localStorage) == "object") {
            localStorage.removeItem(name);
        } else {
            console.log(NOT_SUPPORTED_MSG);
        }
    };

    return {
        persistentWrite : persistentWrite,
        persistentRead : persistentRead,
        persistentRemove : persistentRemove
    };
})();
